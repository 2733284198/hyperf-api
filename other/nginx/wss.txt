upstream websocket {
    # 设置负载均衡模式为 IP Hash 算法模式，这样不同的客户端每次请求都会与同一节点进行交互
    # ip_hash;
    # Hyperf WebSocket Server 的 IP 及 端口
    # server 127.0.0.1:9503;
    server 127.0.0.1:7272;
}

server {
    listen       443 ssl;
    server_name  ws.*.com;

    #ssl                  on;
    ssl_certificate      /usr/local/nginx/ssl/lvcoin.crt;
    ssl_certificate_key  /usr/local/nginx/ssl/lvcoin.key;

    ssl_session_timeout  5m;

    ssl_ciphers ECDHE-RSA-AES128-GCM-SHA256:ECDHE:ECDH:AES:HIGH:!NULL:!aNULL:!MD5:!ADH:!RC4;
    #ssl_ciphers ALL:!ADH:!EXPORT56:!RC4+RSA:+HIGH:+MEDIUM:+LOW:+SSLv2:+EXP;
    ssl_protocols TLSv1 TLSv1.1 TLSv1.2;
    ssl_prefer_server_ciphers on;

    # access_log /home/www/task-server/runtime/lvcoin.log;

    #反向代理配置
    location / {
      # WebSocket Header
        proxy_http_version 1.1;
        proxy_set_header Upgrade websocket;
        proxy_set_header Connection "Upgrade";

        # 将客户端的 Host 和 IP 信息一并转发到对应节点  
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header Host $http_host;

        # 客户端与服务端无交互 60s 后自动断开连接，请根据实际业务场景设置
        # proxy_read_timeout 60s ;

        # 执行代理访问真实服务器
        proxy_pass http://websocket;        
    }
}
